Data Acquisition and Preparation

Gather and clean Nigerian health data from multiple sources

Data Loading and Cleaning 
import pandas as pd
import numpy as np

# Load datasets
facilities_df = pd.read_csv('nigerian_health_facilities.csv')
disease_df = pd.read_csv('nigerian_disease_prevalence.csv')
population_df = pd.read_csv('nigerian_population_data.csv')

# Clean facilities data
facilities_df = facilities_df.dropna(subset=['latitude', 'longitude'])
facilities_df['facility_type'] = facilities_df['facility_type'].fillna('Unknown')

# Merge disease and population data
health_df = pd.merge(disease_df, population_df, on=['state', 'year'])

# Calculate facility density per 100,000 people
facility_counts = facilities_df.groupby('state').size().reset_index(name='facility_count')
health_df = pd.merge(health_df, facility_counts, on='state', how='left')
health_df['facility_density'] = health_df['facility_count'] / health_df['population'] * 100000

# Save cleaned data
health_df.to_csv('nigerian_health_analysis.csv', index=False)

Insight:

Data cleaning is crucial for accurate analysis. We handle missing values, merge datasets, and create new metrics like facility density to enable meaningful comparisons between states.

Exploratory Data Analysis

Use Seaborn and Matplotlib to explore patterns and relationships

Facility Distribution Analysis 
import matplotlib.pyplot as plt
import seaborn as sns

# Set style
sns.set_theme(style="whitegrid", palette="muted")

# Facility distribution by state
plt.figure(figsize=(12, 8))
ax = sns.barplot(
    x='facility_count', 
    y='state',
    data=health_df.sort_values('facility_count', ascending=False).head(10),
    palette="viridis"
)
plt.title('Top 10 Nigerian States by Health Facility Count', fontsize=16)
plt.xlabel('Number of Facilities', fontsize=12)
plt.ylabel('State', fontsize=12)

# Add annotations
for p in ax.patches:
    width = p.get_width()
    plt.text(
        width + 20, 
        p.get_y() + p.get_height()/2,
        '{:1.0f}'.format(width),
        ha='left', 
        va='center'
    )
    
plt.tight_layout()
plt.savefig('facility_distribution.png', dpi=300)
plt.show()

Insight:

This visualization reveals significant disparities in healthcare infrastructure. Lagos has nearly 3 times more health facilities than Rivers state despite similar population sizes.

Disease Prevalence vs Facility Density 
# Custom style for publication
plt.style.use('seaborn-whitegrid')
custom_style = {
    'axes.edgecolor': '#333',
    'axes.labelcolor': '#333',
    'text.color': '#333',
    'font.family': ['DejaVu Sans', 'Arial', 'sans-serif'],
    'grid.linestyle': '--',
    'grid.alpha': 0.3
}
plt.rcParams.update(custom_style)

# Create figure
fig, ax = plt.subplots(figsize=(10, 6))

# Scatter plot with regression line
sns.regplot(
    x='facility_density', 
    y='malaria_rate', 
    data=health_df,
    scatter_kws={'s': 100, 'alpha': 0.7, 'color': '#008751'},
    line_kws={'color': '#d62728', 'linewidth': 2},
    ax=ax
)

# Annotate outliers
outliers = health_df[health_df['malaria_rate'] > 40]
for i, row in outliers.iterrows():
    ax.annotate(row['state'], 
                (row['facility_density'], row['malaria_rate']),
                xytext=(5, 5), textcoords='offset points',
                fontsize=9, color='#d62728')

# Labels and title
ax.set_title('Relationship Between Healthcare Access and Malaria Prevalence', fontsize=14)
ax.set_xlabel('Health Facilities per 100,000 People', fontsize=12)
ax.set_ylabel('Malaria Rate (%)', fontsize=12)

# Add footnote
plt.figtext(0.5, 0.01, 'Data Source: Nigerian Ministry of Health | Visualization: NITDA/NCAIR Cohort 27', 
            ha='center', fontsize=9, color='#666')

plt.tight_layout()
plt.subplots_adjust(bottom=0.15)
plt.savefig('malaria_vs_facilities.png', dpi=300, bbox_inches='tight')
plt.show()

Insight:

This analysis reveals a strong negative correlation between healthcare access and malaria prevalence. States with higher facility density tend to have lower malaria rates, highlighting the importance of healthcare infrastructure investment.

Dashboard Implementation 
import dash
import dash_core_components as dcc
import dash_html_components as html
from dash.dependencies import Input, Output
import plotly.express as px
import pandas as pd

# Load data
df = pd.read_csv('nigerian_health_analysis.csv')

# Initialize Dash app
app = dash.Dash(__name__)

# App layout
app.layout = html.Div([
    html.Div(className="header", children=[
        html.H1("Nigerian Health Analytics Dashboard"),
        html.P("NITDA/NCAIR Data Science Cohort 27")
    ]),
    
    html.Div(className="controls", children=[
        dcc.Dropdown(
            id='state-selector',
            options=[{'label': s, 'value': s} for s in df['state'].unique()],
            value=['Lagos', 'Kano'],
            multi=True,
            placeholder="Select states..."
        ),
        dcc.Dropdown(
            id='disease-selector',
            options=[
                {'label': 'Malaria', 'value': 'malaria_rate'},
                {'label': 'Cholera', 'value': 'cholera_rate'},
                {'label': 'Typhoid', 'value': 'typhoid_rate'}
            ],
            value='malaria_rate',
            placeholder="Select disease..."
        ),
        dcc.RangeSlider(
            id='year-slider',
            min=df['year'].min(),
            max=df['year'].max(),
            value=[df['year'].min(), df['year'].max()],
            marks={str(year): str(year) for year in df['year'].unique()}
        )
    ]),
    
    html.Div(className="dashboard", children=[
        dcc.Graph(id='map-plot'),
        dcc.Graph(id='disease-trend'),
        dcc.Graph(id='facility-comparison')
    ]),
    
    html.Div(className="footer", children=[
        html.P("Data Source: Nigerian Ministry of Health | Â© 2023 NITDA/NCAIR Cohort 27")
    ])
])

# Callback for updating plots
@app.callback(
    [Output('map-plot', 'figure'),
     Output('disease-trend', 'figure'),
     Output('facility-comparison', 'figure')],
    [Input('state-selector', 'value'),
     Input('disease-selector', 'value'),
     Input('year-slider', 'value')]
)
def update_plots(selected_states, selected_disease, years):
    # Filter data based on selections
    filtered_df = df[
        (df['state'].isin(selected_states)) & 
        (df['year'] >= years[0]) & 
        (df['year'] <= years[1])
    ]
    
    # Create map plot
    map_fig = px.scatter_mapbox(
        filtered_df,
        lat='latitude',
        lon='longitude',
        color='state',
        size='facility_count',
        hover_name='state',
        hover_data=['facility_count', 'population', selected_disease],
        zoom=5,
        height=500,
        title='Health Facilities Distribution'
    )
    map_fig.update_layout(mapbox_style="open-street-map")
    
    # Create disease trend plot
    trend_fig = px.line(
        filtered_df,
        x='year',
        y=selected_disease,
        color='state',
        title=f'{selected_disease.replace("_", " ").title()} Trends',
        markers=True
    )
    
    # Create facility comparison
    comp_fig = px.bar(
        filtered_df,
        x='state',
        y=['hospital_count', 'clinic_count', 'pharmacy_count'],
        title='Healthcare Facility Comparison',
        barmode='group',
        labels={'value': 'Number of Facilities'}
    )
    
    return map_fig, trend_fig, comp_fig

if __name__ == '__main__':
    app.run_server(debug=True)
