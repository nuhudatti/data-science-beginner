# Nigerian Screen Time Analysis
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load Nigerian Well-being Dataset
data = pd.read_csv('nigerian_screen_wellbeing.csv')

# Clean and prepare data
data = data.dropna()
data['screen_time'] = data['screen_time'].apply(lambda x: x*60) # Convert to minutes

# Visualization: Screen Time vs Well-being
plt.figure(figsize=(10,6))
sns.scatterplot(x='screen_time', y='wellbeing_score', 
                data=data, hue='region', 
                palette='viridis')
plt.title('Screen Time vs Mental Well-being in Nigeria')
plt.xlabel('Daily Screen Time (minutes)')
plt.ylabel('Well-being Score')
plt.axvline(x=240, color='r', linestyle='--')
plt.text(250, 20, 'WHO Recommended Limit', color='white')
plt.show()
