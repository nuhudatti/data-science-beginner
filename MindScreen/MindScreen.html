Final National-Standard Single-File Script

Your exact, presentation-ready code with no warnings and clear flow.

screen_time_analysis.py
"""
Screen Time Analysis - Final National Standard Single File Version
Author: Nuhu Muhammad Datti
Date: 2025
Description:
    Analyzes the Kaggle screen time dataset.
    Clean, beginner-friendly, and ready for presentation.
"""

# =========================
# 1. Import Required Libraries
# =========================
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# =========================
# 2. Load Dataset
# =========================
file_name = "screen_time.csv"
try:
    df = pd.read_csv(file_name)
    print(f"✅ Successfully loaded '{file_name}'")
except FileNotFoundError:
    print(f"❌ ERROR: '{file_name}' not found. Place it in the same folder as this script.")
    raise SystemExit

# =========================
# 3. Explore Dataset
# =========================
print("\n--- Dataset Preview ---")
print(df.head())

print("\n--- Dataset Info ---")
print(df.info())

print("\n--- Summary Statistics ---")
print(df.describe())

# =========================
# 4. Data Cleaning (Future-proof, no warnings)
# =========================
print("\n--- Data Cleaning ---")
df = df.drop_duplicates()
for column in df.columns:
    if df[column].dtype in ["int64", "float64", "int32", "float32"]:
        df[column] = df[column].fillna(df[column].mean())
    else:
        df[column] = df[column].fillna(df[column].mode()[0])
print("✅ Missing values handled.")
print("✅ Duplicates removed.")

# =========================
# 5. Basic Analysis
# =========================
print("\n--- Basic Analysis ---")
avg_screen_time = df['daily_screen_time'].mean()
print(f"📊 Average daily screen time: {avg_screen_time:.2f} hours")
avg_sleep = df['sleep_duration'].mean()
print(f"💤 Average sleep duration: {avg_sleep:.2f} hours")
top_gender = df['gender'].mode()[0]
print(f"🧍 Most common gender: {top_gender}")
stress_counts = df['self_reported_stress'].value_counts()
print("\n📌 Stress level distribution:\n", stress_counts)

# =========================
# 6. Visualizations
# =========================
print("\n--- Generating Visualizations ---")
sns.set(style="whitegrid")

# 6.1 Stress Level Distribution
plt.figure(figsize=(8, 5))
sns.countplot(data=df, x='self_reported_stress')
plt.title("Stress Level Distribution")
plt.xlabel("Stress Level")
plt.ylabel("Count")
plt.tight_layout()
plt.show()

# 6.2 Age vs Daily Screen Time by Stress Level
plt.figure(figsize=(8, 5))
sns.scatterplot(data=df, x='age', y='daily_screen_time', hue='self_reported_stress')
plt.title("Age vs Daily Screen Time by Stress Level")
plt.xlabel("Age")
plt.ylabel("Daily Screen Time (hours)")
plt.tight_layout()
plt.show()

# 6.3 Correlation Heatmap
plt.figure(figsize=(10, 8))
corr = df.corr(numeric_only=True)
sns.heatmap(corr, annot=True, cmap='coolwarm')
plt.title("Correlation Matrix")
plt.tight_layout()
plt.show()

# 6.4 Screen Time vs Stress Levels (jittered)
plt.figure(figsize=(7, 5))
sns.stripplot(data=df, x="self_reported_stress", y="daily_screen_time", jitter=True)
plt.title("Screen Time vs Stress Levels")
plt.xlabel("Stress Level")
plt.ylabel("Daily Screen Time (hours)")
plt.tight_layout()
plt.show()

# =========================
# 7. Save Cleaned Data
# =========================
clean_file_name = "screen_time_cleaned.csv"
df.to_csv(clean_file_name, index=False)
print(f"💾 Cleaned data saved as '{clean_file_name}'")

# =========================
# 8. End of Script
# =========================
print("\n✅ Analysis Complete. Ready for presentation!")
Tip: Put screen_time.csv next to the script, then run: python screen_time_analysis.py
